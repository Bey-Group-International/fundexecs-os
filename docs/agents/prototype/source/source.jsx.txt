function nextStage(list, s) { const i = list.indexOf(s); return i < list.length - 1 ? list[i + 1] : s; }
function fitColor(f) { return f >= 85 ? 'var(--success)' : f >= 75 ? 'var(--gold-1)' : 'var(--fg-3)'; }

function LpCapitalMap({ lps, target, noun, onRunLp, onRun, fitColor }) {
  const [openLp, setOpenLp] = useState(null);
  const committed = lps.filter(l => l.stage === 'Committed').reduce((s, l) => s + l.amt, 0);
  const soft = lps.filter(l => l.stage === 'Soft-circled').reduce((s, l) => s + l.amt, 0);
  const pipeline = lps.filter(l => l.stage === 'Target' || l.stage === 'Contacted').reduce((s, l) => s + l.amt, 0);
  const pctC = Math.min(100, Math.round((committed / target) * 100));
  const pctS = Math.min(100 - pctC, Math.round((soft / target) * 100));
  const stages = ['Target', 'Contacted', 'Soft-circled', 'Committed'];
  const byStage = (st) => lps.filter(l => l.stage === st);
  const lp = openLp && lps.find(l => l.id === openLp);

  return (
    <Panel title="LP Capital Map" eyebrow="Fit-scored & ranked by Sloane · tap an LP to open" icon="landmark"
      action={<Button variant="ghost" size="sm" icon="sparkles" onClick={() => onRun(ASK_EARN)}>Build more targets</Button>}>
      {/* raise thermometer */}
      <div style={{ marginBottom: 16 }}>
        <div style={{ display: 'flex', alignItems: 'baseline', justifyContent: 'space-between', marginBottom: 7 }}>
          <span style={{ fontSize: 12, color: 'var(--fg-3)' }}><b style={{ color: 'var(--fg-1)' }}>${committed}M</b> committed <span style={{ color: 'var(--fg-5)' }}>+ ${soft}M soft-circled</span></span>
          <span style={{ fontSize: 12, color: 'var(--fg-4)' }}>of <b style={{ color: 'var(--gold-1)' }}>${target}M</b> target</span>
        </div>
        <div style={{ height: 12, borderRadius: 999, background: 'var(--surface-2)', overflow: 'hidden', display: 'flex', border: '1px solid var(--border)' }}>
          <div style={{ width: pctC + '%', background: 'linear-gradient(90deg,#1F8A5B,#2fae74)', transition: 'width .4s' }} />
          <div style={{ width: pctS + '%', background: 'linear-gradient(90deg,#F7C948,#E5A823)', opacity: 0.55, transition: 'width .4s' }} />
        </div>
        <div style={{ fontSize: 11, color: 'var(--fg-5)', marginTop: 6 }}>{pctC}% closed · {Math.round((pipeline / target) * 100)}% more in pipeline</div>
      </div>
      {/* stage funnel */}
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(4,1fr)', gap: 8, marginBottom: 16 }}>
        {stages.map(st => {
          const items = byStage(st); const amt = items.reduce((s, l) => s + l.amt, 0);
          return (
            <div key={st} className="card-flat" style={{ padding: '10px 11px', background: 'var(--surface-1)', borderTop: `2px solid ${TONE[LP_STAGE_TONE[st]].c}` }}>

              <div style={{ display: 'flex', alignItems: 'baseline', gap: 5, marginTop: 4 }}>
                <span className="t-metric" style={{ fontSize: 18 }}>{items.length}</span>
                <span style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>· ${amt}M</span>
              </div>
            </div>
          );
        })}
      </div>
      {/* LP cards */}
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill,minmax(260px,1fr))', gap: 10 }}>
        {lps.slice().sort((a, b) => stages.indexOf(b.stage) - stages.indexOf(a.stage) || b.fit - a.fit).map(l => (
          <button key={l.id} className="clickable card-flat" onClick={() => setOpenLp(l.id)} style={{ textAlign: 'left', padding: '13px 14px', background: 'var(--surface-1)' }}>
            <div style={{ display: 'flex', alignItems: 'center', gap: 10 }}>

                <div style={{ fontSize: 13, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{l.name}</div>
                <div style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>{l.type}</div>
              </div>
              <Badge tone={LP_STAGE_TONE[l.stage]} style={{ fontSize: 9.5, padding: '2px 7px' }}>{l.stage}</Badge>
            </div>
            <div style={{ display: 'flex', alignItems: 'center', gap: 14, marginTop: 11 }}>
              <span style={{ fontSize: 11, color: 'var(--fg-4)' }}>Check <b className="t-mono" style={{ color: 'var(--fg-2)' }}>${l.amt}M</b></span>

              <div style={{ display: 'grid', gridTemplateColumns: 'repeat(3,1fr)', gap: 10 }}>

              </div>

function StatTile({ label, value, sub, tone = 'gold' }) {
  return (
    <div className="card-flat" style={{ padding: '12px 14px', background: 'var(--surface-1)' }}>
      <div style={{ fontSize: 11, color: 'var(--fg-4)' }}>{label}</div>
      <div className="t-metric" style={{ fontSize: 21, marginTop: 6, color: TONE[tone].c }}>{value}</div>
      {sub && <div style={{ fontSize: 10.5, color: 'var(--fg-5)', marginTop: 2 }}>{sub}</div>}
    </div>
  );
}

function SourceHub({ mandate, group, pct, onRun }) {
  const [tab, setTab] = useState('lps');
  const [lps, setLps] = useState(LP_SEED);
  const [deals, setDeals] = useState(DEAL_SEED);
  const [providers, setProviders] = useState(PROVIDER_SEED);
  const [leads, setLeads] = useState(LEAD_SEED);
  const noun = SRC_NOUN[group] || 'LP';

  const committed = lps.filter(l => l.stage === 'Committed').reduce((s, l) => s + l.amt, 0);
  const soft = lps.filter(l => l.stage === 'Soft-circled').reduce((s, l) => s + l.amt, 0);
  const advanced = lps.filter(l => l.stage !== 'Target').length;

  function runLp(lp) {
    const act = LP_NEXT[lp.stage]; if (!act) return;
    onRun({ id: 'lp-' + lp.id + '-' + lp.stage, hub: 'source', xp: 20, title: `${act} — ${lp.name}`,
      apply: () => setLps(prev => prev.map(x => x.id === lp.id ? { ...x, stage: nextStage(LP_STAGES, x.stage) } : x)),
      run: { title: `${act} — ${lp.name}`, steps: ['Pull engagement + fit signals', `Draft the ${act.toLowerCase()}`, 'Attach your one-pager + track record', 'Prepare for your approval'],

  }
  function runDeal(d) {
    const ns = nextStage(DEAL_STAGES, d.stage); if (ns === d.stage) return;
    onRun({ id: 'deal-' + d.id + '-' + d.stage, hub: 'source', xp: 20, title: `Advance ${d.name} → ${ns}`,
      apply: () => setDeals(prev => prev.map(x => x.id === d.id ? { ...x, stage: ns } : x)),
      run: { title: `Advance ${d.name} → ${ns}`, steps: ['Pull the latest on the deal', `Prepare the ${ns} package`, 'Cross-check against your thesis', 'Prepare for your approval'],
        draftTitle: `${d.name} → ${ns}`, draft: `Earn assembled what's needed to move ${d.name} (${d.sector}, score ${d.score}) into ${ns}. Approve to advance it on your pipeline.` } });

      run: { title: `${act} — ${p.name}`, steps: ['Pull vetting + fit on ' + p.cat, `Draft the ${act.toLowerCase()}`, 'Check terms & references', 'Prepare for your approval'],

  }
  function runLead(l) {
    const act = LEAD_NEXT[l.stage]; if (!act) return;
    onRun({ id: 'lead-' + l.id + '-' + l.stage, hub: 'source', xp: 12, title: `${act} — ${l.name}`,
      apply: () => setLeads(prev => prev.map(x => x.id === l.id ? { ...x, stage: nextStage(LEAD_STAGES, x.stage) } : x)),
      run: { title: `${act} — ${l.name}`, steps: ['Pull intent + firmographics', `Draft the ${act.toLowerCase()}`, 'Personalize to their segment', 'Prepare for your approval'],
        draftTitle: `${act} — ${l.name}`, draft: `Earn prepared the ${act.toLowerCase()} for ${l.name} (${l.seg}, intent ${l.intent}) to drive revenue for your portfolio company. Approve to send and advance the lead.` } });
  }

  return (
    <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
      {/* raise summary */}
      <div className="card" style={{ padding: '18px 20px' }}>
        <div style={{ display: 'flex', alignItems: 'center', gap: 13 }}>
          <span style={{ width: 44, height: 44, borderRadius: 12, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--accent-soft)', border: '1px solid var(--accent-line)', color: 'var(--accent)' }}><Icon name="radar" size={22} /></span>
          <div style={{ flex: 1, minWidth: 0 }}>
            <div className="t-h2" style={{ fontSize: 19 }}>Source</div>
            <div style={{ fontSize: 12.5, color: 'var(--fg-3)', marginTop: 2 }}>The team finds your capital and deals. Approve a move and it advances here.</div>
          </div>
          <div style={{ textAlign: 'right', flex: 'none' }}>
            <div className="t-metric" style={{ fontSize: 22, color: 'var(--gold-1)' }}>{pct}%</div>
            <div style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>Source ready</div>
          </div>
        </div>
        <div style={{ marginTop: 14, marginBottom: 16 }}><ProgressBar value={pct} gradient="linear-gradient(90deg,#F7C948,#E5A823)" height={6} /></div>
        <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit,minmax(130px,1fr))', gap: 10 }}>
          <StatTile label="Committed" value={`$${committed}M`} sub="closed allocations" tone="success" />
          <StatTile label="Soft-circled" value={`$${soft}M`} sub="near-term" tone="gold" />
          <StatTile label={`${noun === 'LP' ? 'LPs' : noun + 's'} advanced`} value={`${advanced}/${lps.length}`} sub="past first touch" tone="azure" />

        </div>
      </div>

      <div style={{ overflowX: 'auto', overflowY: 'hidden', paddingBottom: 2 }}><SegTabs active={tab} onChange={setTab} tabs={[{ id: 'lps', label: SRC_TITLE[group] || 'LP Capital Map', icon: 'landmark' }, { id: 'deals', label: 'Deal pipeline', icon: 'trending-up' }, { id: 'partners', label: 'Partner Network', icon: 'handshake' }, { id: 'leads', label: 'Lead Engine', icon: 'megaphone' }]} /></div>

          : (
        <Panel title={SRC_TITLE[group] || 'LP Capital Map'} eyebrow="Fit-scored & ranked by Sloane" icon="landmark"
          action={<Button variant="ghost" size="sm" icon="sparkles" onClick={() => onRun(ASK_EARN)}>Build more targets</Button>}>
          <div style={{ display: 'grid', gridTemplateColumns: '1.6fr 0.8fr 0.6fr 1fr auto', gap: 10, padding: '0 4px 8px', alignItems: 'center' }} className="t-eyebrow">
            <span>Investor</span><span>Check</span><span>Fit</span><span>Stage</span><span></span>
          </div>
          <div style={{ display: 'flex', flexDirection: 'column', gap: 6 }}>
            {lps.map(lp => (
              <div key={lp.id} className="card-flat" style={{ display: 'grid', gridTemplateColumns: '1.6fr 0.8fr 0.6fr 1fr auto', gap: 10, alignItems: 'center', padding: '10px 13px', background: 'var(--surface-1)' }}>
                <div style={{ minWidth: 0 }}>
                  <div style={{ fontSize: 13, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{lp.name}</div>
                  <div style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>{lp.type} · {lp.warmth}</div>
                </div>
                <span className="t-mono" style={{ fontSize: 12.5, color: 'var(--fg-2)' }}>${lp.amt}M</span>
                <span style={{ fontSize: 13, fontWeight: 600, color: fitColor(lp.fit) }}>{lp.fit}</span>
                <Badge tone={LP_STAGE_TONE[lp.stage]} style={{ fontSize: 10, padding: '2px 8px' }}>{lp.stage}</Badge>
                {lp.stage === 'Committed'
                  ? <span style={{ display: 'inline-flex', alignItems: 'center', gap: 5, fontSize: 11.5, color: 'var(--success)', fontWeight: 600, justifySelf: 'end' }}><Icon name="check-circle-2" size={14} />Closed</span>
                  : <Button variant="secondary" size="sm" icon="sparkles" style={{ justifySelf: 'end' }} onClick={() => runLp(lp)}>{LP_NEXT[lp.stage]}</Button>}
              </div>
            ))}
          </div>
        </Panel>
          )
      ) : tab === 'deals' ? (
        <DealPipeline deals={deals} onRunDeal={runDeal} onRun={onRun} />
      ) : tab === 'partners' ? (
        <PartnerNetwork providers={providers} onRunProvider={runProvider} onRun={onRun} />
      ) : (
        <LeadEngine leads={leads} onRunLead={runLead} onRun={onRun} />
      )}

      <div className="card" style={{ padding: '14px 16px', display: 'flex', alignItems: 'center', gap: 12, background: 'var(--gold-soft)', borderColor: 'var(--gold-line)' }}>
        <EarnCoin size={26} />
        <div style={{ flex: 1, fontSize: 12.5, color: 'var(--fg-2)', lineHeight: 1.5 }}><b style={{ color: 'var(--gold-1)' }}>Earn:</b> I rank every {noun} and deal by fit and momentum. Tap any move — I draft it, you approve, it advances.</div>
      </div>
    </div>
  );
}
