function StepStatus(status) {
  return status === 'signed' || status === 'wired' ? { tone: 'success', label: 'Done', icon: 'check-circle-2' }
    : status === 'ready' ? { tone: 'gold', label: 'Ready', icon: 'pen-line' }
    : { tone: 'neutral', label: 'Waiting', icon: 'circle-dashed' };
}

function SignaturesWires({ sigs, setSigs, wires, setWires, onRun }) {
  const [view, setView] = useState('signatures');
  const sigDone = sigs.filter(s => s.status === 'signed').length;
  const sigPending = sigs.filter(s => s.status !== 'signed');
  const outStaged = wires.filter(w => w.dir === 'out' && w.status === 'staged');
  const inExpected = wires.filter(w => w.dir === 'in' && w.status === 'expected').reduce((s, w) => s + w.amt, 0);
  const outTotal = wires.filter(w => w.dir === 'out').reduce((s, w) => s + w.amt, 0);

  function signOne(sig) {
    onRun({ id: 'sig-' + sig.id, hub: 'execute', xp: 12, title: `Sign — ${sig.doc}`, apply: () => setSigs(prev => prev.map(x => x.id === sig.id ? { ...x, status: 'signed' } : x)),
      run: { title: `Sign — ${sig.doc}`, steps: ['Open the execution copy', 'Apply your signature', 'Notify counterparties', 'Log to Chain of Trust'], draftTitle: sig.doc, draft: `Your signature on ${sig.doc} (${sig.role}) ${sig.drives.toLowerCase()}. Approve to e-sign — it's timestamped and logged.` } });
  }
  function sendWire(w) {
    onRun({ id: 'wire-' + w.id, hub: 'execute', xp: 16, title: `${w.dir === 'out' ? 'Send' : 'Confirm'} — ${w.label}`, apply: () => setWires(prev => prev.map(x => x.id === w.id ? { ...x, status: 'cleared', when: 'Cleared' } : x)),
      run: { title: `${w.dir === 'out' ? 'Send wire' : 'Confirm receipt'} — ${w.label}`, steps: ['Verify account & amount', 'Dual-control approval', w.dir === 'out' ? 'Release the wire' : 'Match the inbound funds', 'Log to Chain of Trust'], draftTitle: `$${w.amt}M · ${w.label}`, draft: `${w.dir === 'out' ? 'Outbound' : 'Inbound'} wire of $${w.amt}M — ${w.to}. ${w.drives}. Approve to ${w.dir === 'out' ? 'release' : 'confirm'} under dual control.` } });
  }

  return (
    <Panel title="Signatures & wires" eyebrow="The signature room & money movement" icon="banknote"
      action={<Button variant="ghost" size="sm" icon="sparkles" onClick={() => onRun(ASK_EARN)}>Ask Earn</Button>}>
      {/* summary */}
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit,minmax(140px,1fr))', gap: 10, marginBottom: 14 }}>
        <div className="card-flat" style={{ padding: '12px 14px', background: 'var(--surface-1)' }}><div style={{ fontSize: 10.5, color: 'var(--fg-4)' }}>Signatures</div><div className="t-metric" style={{ fontSize: 19, marginTop: 4, color: sigPending.length ? 'var(--warning)' : 'var(--success)' }}>{sigDone}/{sigs.length}</div><div style={{ fontSize: 10, color: 'var(--fg-5)' }}>{sigPending.length} awaiting</div></div>
        <div className="card-flat" style={{ padding: '12px 14px', background: 'var(--surface-1)' }}><div style={{ fontSize: 10.5, color: 'var(--fg-4)' }}>Outbound staged</div><div className="t-metric" style={{ fontSize: 19, marginTop: 4, color: 'var(--gold-1)' }}>{outStaged.length}</div><div style={{ fontSize: 10, color: 'var(--fg-5)' }}>${outTotal.toFixed(1)}M total out</div></div>
        <div className="card-flat" style={{ padding: '12px 14px', background: 'var(--surface-1)' }}><div style={{ fontSize: 10.5, color: 'var(--fg-4)' }}>Inbound expected</div><div className="t-metric" style={{ fontSize: 19, marginTop: 4, color: 'var(--azure)' }}>${inExpected.toFixed(1)}M</div><div style={{ fontSize: 10, color: 'var(--fg-5)' }}>capital incoming</div></div>
      </div>
      <SegTabs active={view} onChange={setView} tabs={[{ id: 'signatures', label: 'Signatures', icon: 'pen-line' }, { id: 'wires', label: 'Wire transfers', icon: 'banknote' }, { id: 'accounts', label: 'Accounts', icon: 'landmark' }]} />
      <div style={{ height: 12 }} />
      {view === 'signatures' ? (
        <div style={{ display: 'flex', flexDirection: 'column', gap: 7 }}>
          {sigs.slice().sort((a, b) => (a.status === 'signed') - (b.status === 'signed')).map(s => {
            const st = SIG_STATUS[s.status]; const done = s.status === 'signed';
            return (
              <div key={s.id} className="card-flat" style={{ display: 'flex', alignItems: 'center', gap: 12, padding: '12px 14px', background: 'var(--surface-1)', borderLeft: `2px solid ${TONE[st.tone].c}`, opacity: done ? 0.74 : 1 }}>
                <span style={{ width: 32, height: 32, borderRadius: 9, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: done ? 'var(--success-soft)' : 'var(--surface-2)', border: `1px solid ${done ? 'var(--success-line)' : 'var(--border)'}`, color: done ? 'var(--success)' : 'var(--fg-3)' }}><Icon name={done ? 'check' : 'file-signature'} size={15} /></span>
                <div style={{ flex: 1, minWidth: 0 }}>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
                    <span style={{ fontSize: 13, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{s.doc}</span>
                    {s.amt && <span className="t-mono" style={{ fontSize: 11, color: 'var(--gold-1)', flex: 'none' }}>{s.amt}</span>}
                  </div>
                  <div style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>{s.party} · {s.role} · {s.drives}</div>
                </div>
                <Badge tone={st.tone} style={{ fontSize: 9, padding: '1px 6px', flex: 'none' }}>{st.label}</Badge>
                {s.status === 'pending' && <Button variant="gold" size="sm" icon="pen-line" style={{ flex: 'none' }} onClick={() => signOne(s)}>Sign</Button>}
                {s.status === 'partial' && <Button variant="secondary" size="sm" icon="send" style={{ flex: 'none' }} onClick={() => signOne(s)}>Chase</Button>}
                {done && <Icon name="lock" size={14} color="var(--success)" style={{ flex: 'none' }} />}
              </div>
            );
          })}
        </div>
      ) : view === 'wires' ? (
        <div style={{ display: 'flex', flexDirection: 'column', gap: 7 }}>
          {wires.slice().sort((a, b) => (a.status === 'cleared') - (b.status === 'cleared')).map(w => {
            const st = WIRE_STATUS[w.status]; const done = w.status === 'cleared'; const out = w.dir === 'out';
            return (
              <div key={w.id} className="card-flat" style={{ display: 'flex', alignItems: 'center', gap: 12, padding: '12px 14px', background: 'var(--surface-1)', borderLeft: `2px solid ${TONE[st.tone].c}`, opacity: done ? 0.74 : 1 }}>
                <span style={{ width: 32, height: 32, borderRadius: 9, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: out ? 'var(--warning-soft)' : 'var(--azure-soft)', border: `1px solid ${out ? 'var(--warning-line)' : 'var(--azure-line)'}`, color: out ? 'var(--warning)' : 'var(--azure)' }}><Icon name={out ? 'arrow-up-right' : 'arrow-down-left'} size={15} /></span>
                <div style={{ flex: 1, minWidth: 0 }}>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
                    <span style={{ fontSize: 13, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{w.label}</span>
                    <span className="t-mono" style={{ fontSize: 12.5, fontWeight: 600, color: out ? 'var(--fg-1)' : 'var(--success)', flex: 'none' }}>{out ? '−' : '+'}${w.amt}M</span>
                  </div>
                  <div style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>{w.to} · {w.drives}</div>
                </div>
                <Badge tone={st.tone} style={{ fontSize: 9, padding: '1px 6px', flex: 'none' }}>{st.label}</Badge>
                {w.status === 'staged' && <Button variant="gold" size="sm" icon="send" style={{ flex: 'none' }} onClick={() => sendWire(w)}>Release</Button>}
                {w.status === 'expected' && <Button variant="secondary" size="sm" icon="check" style={{ flex: 'none' }} onClick={() => sendWire(w)}>Confirm</Button>}
                {done && <Icon name="lock" size={14} color="var(--success)" style={{ flex: 'none' }} />}
              </div>
            );
          })}
          <div style={{ display: 'flex', alignItems: 'center', gap: 8, marginTop: 8, fontSize: 11, color: 'var(--fg-5)' }}><Icon name="shield-check" size={13} color="var(--success)" />Every wire moves under dual-control approval and logs to your Chain of Trust.</div>
        </div>
      ) : (
        <div style={{ display: 'flex', flexDirection: 'column', gap: 10 }}>
          <div className="card-flat" style={{ padding: '14px 16px', background: 'var(--surface-1)', display: 'flex', alignItems: 'baseline', justifyContent: 'space-between' }}>
            <div><div style={{ fontSize: 11, color: 'var(--fg-4)' }}>Total fund cash</div><div className="t-metric" style={{ fontSize: 24, marginTop: 3, color: 'var(--success)' }}>${EX_ACCOUNTS.reduce((s, a) => s + a.balance, 0).toFixed(1)}M</div></div>
            <span style={{ fontSize: 11, color: 'var(--fg-5)' }}>across {EX_ACCOUNTS.length} accounts</span>
          </div>
          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit,minmax(230px,1fr))', gap: 10 }}>
            {EX_ACCOUNTS.map(a => {
              const neg = a.balance < 0;
              return (
                <div key={a.id} className="card-flat" style={{ padding: '13px 14px', background: 'var(--surface-1)' }}>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 10 }}>
                    <span style={{ width: 30, height: 30, borderRadius: 8, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--surface-2)', border: '1px solid var(--border)', color: 'var(--fg-3)' }}><Icon name={a.icon} size={15} /></span>
                    <div style={{ flex: 1, minWidth: 0 }}><div style={{ fontSize: 12.5, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{a.name}</div><div style={{ fontSize: 10, color: 'var(--fg-5)' }}>{a.bank}</div></div>
                  </div>
                  <div className="t-metric" style={{ fontSize: 19, marginTop: 10, color: neg ? 'var(--warning)' : 'var(--fg-1)' }}>{neg ? '−' : ''}${Math.abs(a.balance).toFixed(1)}M</div>
                  <div style={{ fontSize: 10.5, color: 'var(--fg-5)', marginTop: 2 }}>{a.note}</div>
                </div>

            })}
          </div>
          <div style={{ display: 'flex', alignItems: 'center', gap: 8, fontSize: 11, color: 'var(--fg-5)' }}><Icon name="refresh-cw" size={13} color="var(--fg-4)" />Balances sync from your banks. Earn flags when a call or wire is needed.</div>

              <div style={{ display: 'flex', alignItems: 'center', gap: 9, marginBottom: 9, flexWrap: 'wrap' }}>
                <span style={{ display: 'inline-flex', alignItems: 'center', gap: 6, fontSize: 12.5, fontWeight: 700, color: closed ? 'var(--success)' : 'var(--gold-1)', background: closed ? 'var(--success-soft)' : 'var(--gold-soft)', border: `1px solid ${closed ? 'var(--success-line)' : 'var(--gold-line)'}`, borderRadius: 999, padding: '4px 12px' }}><Icon name={closed ? 'circle-check-big' : 'pen-line'} size={14} />{closed ? 'Closed & funded' : `${steps.length - doneCount} steps to close`}</span>
                <span style={{ fontSize: 11.5, color: 'var(--fg-4)' }}>{cl.counterparty}</span>
              </div>
              <div style={{ display: 'flex', alignItems: 'baseline', justifyContent: 'space-between', marginBottom: 5 }}>
                <span style={{ fontSize: 11.5, color: 'var(--fg-3)' }}><b style={{ color: 'var(--fg-1)' }}>{doneCount}/{steps.length}</b> steps executed</span>
                <span style={{ fontSize: 11, color: 'var(--fg-4)' }}>{cl.amount} consideration</span>
              </div>
              <ProgressBar value={closePct} gradient="linear-gradient(90deg,#1F8A5B,#2fae74)" height={7} />
            </div>
            <div style={{ display: 'flex', flexDirection: 'column', gap: 8, justifyContent: 'center' }}>
              <Button variant={closed ? 'success' : 'gold'} size="sm" icon={closed ? 'check-circle-2' : 'pen-line'} disabled={closed || nextIdx < 0} onClick={() => nextIdx >= 0 && runStep(steps[nextIdx])}>{closed ? 'Closed' : steps[nextIdx] ? steps[nextIdx].action : '—'}</Button>
              <Button variant="ghost" size="sm" icon="file-text">Closing binder</Button>
            </div>
          </div>
          {/* execution timeline */}
          <div style={{ display: 'flex', flexDirection: 'column' }}>
            {steps.map((s, i) => {
              const st = StepStatus(s.status); const isDone = s.status === 'signed' || s.status === 'wired'; const isNext = i === nextIdx;
              return (
                <div key={s.id} style={{ display: 'flex', gap: 13, alignItems: 'stretch' }}>
                  {/* rail */}
                  <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', flex: 'none', width: 28 }}>
                    <div style={{ width: 28, height: 28, borderRadius: 999, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: isDone ? 'var(--success-soft)' : isNext ? 'var(--gold-soft)' : 'var(--surface-2)', border: `1px solid ${isDone ? 'var(--success-line)' : isNext ? 'var(--gold-line)' : 'var(--border)'}`, color: isDone ? 'var(--success)' : isNext ? 'var(--gold-1)' : 'var(--fg-5)' }}>
                      {isDone ? <Icon name="check" size={14} /> : <span style={{ fontSize: 11, fontWeight: 700 }}>{i + 1}</span>}
                    </div>
                    {i < steps.length - 1 && <div style={{ width: 2, flex: 1, minHeight: 16, background: isDone ? 'var(--success-line)' : 'var(--border)' }} />}
                  </div>
                  {/* card */}
                  <div className="clickable card-flat" onClick={() => setOpenId(s.id)} style={{ flex: 1, textAlign: 'left', display: 'flex', alignItems: 'center', gap: 12, padding: '12px 14px', marginBottom: 10, background: 'var(--surface-1)', border: isNext ? '1px solid var(--gold-line)' : '1px solid var(--border)', cursor: 'pointer', opacity: isDone ? 0.78 : 1 }}>
                    <div style={{ flex: 1, minWidth: 0 }}>
                      <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
                        <span style={{ fontSize: 13, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{s.name}</span>
                        <Badge tone={st.tone} style={{ fontSize: 9, padding: '1px 6px', flex: 'none' }}>{st.label}</Badge>
                      </div>
                      <div style={{ fontSize: 11, color: isNext ? 'var(--gold-1)' : 'var(--fg-5)', marginTop: 3 }}>{s.drives} · {s.who}</div>
                    </div>
                    {isNext && s.action && <Button variant="gold" size="sm" icon="sparkles" style={{ flex: 'none' }} onClick={(e) => { e.stopPropagation(); runStep(s); }}>{s.action}</Button>}
                    {isDone && <Icon name="lock" size={14} color="var(--success)" style={{ flex: 'none' }} />}
                  </div>
                </div>
              );
            })}
          </div>
          {closed && (
            <div className="card-flat fade-in" style={{ padding: '16px 18px', marginTop: 4, background: 'var(--success-soft)', border: '1px solid var(--success-line)', textAlign: 'center' }}>
              <Icon name="circle-check-big" size={28} color="var(--success)" />
              <div className="t-h3" style={{ fontSize: 16, marginTop: 8 }}>{cl.name} is closed.</div>
              <div style={{ fontSize: 12.5, color: 'var(--fg-3)', marginTop: 4 }}>{cl.amount} funded · signed, wired and logged to your Chain of Trust. {firm} now owns it.</div>
            </div>
          )}
        </Panel>
      ) : tab === 'wires' ? (
        <SignaturesWires sigs={sigs} setSigs={setSigs} wires={wires} setWires={setWires} onRun={onRun} />
      ) : tab === 'calls' ? (
        <CapitalCalls calls={calls} setCalls={setCalls} onRun={onRun} />
      ) : (
        <ChainOfTrust onRun={onRun} />
      )}

      {/* step drawer */}
      {it && (
        <>
          <div onClick={() => setOpenId(null)} style={{ position: 'fixed', inset: 0, background: 'rgba(3,6,12,0.64)', backdropFilter: 'blur(3px)', zIndex: 60 }} />
          <div className="fade-in" style={{ position: 'fixed', top: 0, right: 0, bottom: 0, width: 440, maxWidth: '94vw', background: 'var(--bg-2)', borderLeft: '1px solid var(--border-strong)', zIndex: 61, overflowY: 'auto', boxShadow: '-30px 0 80px -30px rgba(0,0,0,0.7)' }}>
            <div style={{ padding: '18px 20px', borderBottom: '1px solid var(--border)', display: 'flex', alignItems: 'center', gap: 12 }}>
              <span style={{ width: 42, height: 42, borderRadius: 11, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--surface-2)', border: '1px solid var(--border)', color: 'var(--fg-2)' }}><Icon name="file-signature" size={20} /></span>
              <div style={{ flex: 1, minWidth: 0 }}>
                <div style={{ fontSize: 15, fontWeight: 600 }}>{it.name}</div>
                <div style={{ fontSize: 11.5, color: 'var(--fg-4)' }}>{it.party} · {it.who}</div>
              </div>
              <button className="clickable" onClick={() => setOpenId(null)} style={{ width: 30, height: 30, borderRadius: 8, color: 'var(--fg-4)', display: 'flex', alignItems: 'center', justifyContent: 'center' }}><Icon name="x" size={17} /></button>
            </div>
            <div style={{ padding: 20, display: 'flex', flexDirection: 'column', gap: 16 }}>
              {(() => { const st = StepStatus(it.status); return (
                <span style={{ display: 'inline-flex', alignItems: 'center', gap: 6, alignSelf: 'flex-start', fontSize: 12, fontWeight: 600, color: TONE[st.tone].c, background: TONE[st.tone].bg, border: `1px solid ${TONE[st.tone].b}`, borderRadius: 999, padding: '4px 11px' }}><Icon name={st.icon} size={13} />{st.label}</span>
              ); })()}
              <div style={{ display: 'flex', alignItems: 'center', gap: 9, padding: '11px 13px', borderRadius: 11, background: 'var(--gold-soft)', border: '1px solid var(--gold-line)' }}>
                <Icon name="zap" size={15} color="var(--gold-1)" /><span style={{ fontSize: 12, color: 'var(--fg-2)' }}><b style={{ color: 'var(--gold-1)' }}>Why it matters:</b> {it.drives}.</span>
              </div>
              <div style={{ fontSize: 12.5, color: 'var(--fg-3)', lineHeight: 1.6 }}>{it.detail}</div>
              <div style={{ display: 'flex', alignItems: 'center', gap: 9, padding: '11px 13px', borderRadius: 11, background: 'var(--surface-1)', border: '1px solid var(--border-faint)' }}>
                <Icon name="shield-check" size={15} color="var(--success)" /><span style={{ fontSize: 11.5, color: 'var(--fg-3)' }}>Every action here is recorded to your 4-layer Chain of Trust — Proof of Truth, Concept, Execution and Work.</span>
              </div>
              {(it.status === 'signed' || it.status === 'wired') ? (
                <div style={{ display: 'flex', alignItems: 'center', gap: 9, padding: '14px 15px', borderRadius: 13, background: 'var(--success-soft)', border: '1px solid var(--success-line)', color: 'var(--success)', fontSize: 13, fontWeight: 600 }}><Icon name="lock" size={17} />Executed · recorded to Chain of Trust</div>
              ) : it.status === 'ready' ? (
                <div style={{ padding: '14px 15px', borderRadius: 13, background: 'var(--gold-soft)', border: '1px solid var(--gold-line)' }}>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 9, marginBottom: 8 }}><EarnCoin size={24} /><span style={{ fontSize: 12.5, fontWeight: 600, color: 'var(--gold-1)' }}>Ready to execute</span></div>
                  <div style={{ fontSize: 12, color: 'var(--fg-2)', lineHeight: 1.5, marginBottom: 11 }}>This step is ready. {it.action} — nothing executes until you approve, and it's logged the moment you do.</div>
                  <Button variant="gold" size="sm" icon="pen-line" style={{ width: '100%' }} onClick={() => { runStep(it); setOpenId(null); }}>{it.action}</Button>
                </div>
              ) : (
                <div style={{ display: 'flex', alignItems: 'center', gap: 9, padding: '14px 15px', borderRadius: 13, background: 'var(--surface-1)', border: '1px solid var(--border)', color: 'var(--fg-4)', fontSize: 12.5 }}><Icon name="clock" size={16} />Waiting on the previous step</div>
              )}
            </div>
          </div>
        </>
      )}
    </div>
  );
}
