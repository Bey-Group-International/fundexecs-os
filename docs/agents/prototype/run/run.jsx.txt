              <ProgressBar value={ddPct} gradient="linear-gradient(90deg,#1F8A5B,#2fae74)" height={7} />
            </div>
            <div style={{ display: 'flex', flexDirection: 'column', gap: 8, justifyContent: 'center' }}>
              <Button variant={icReady ? 'gold' : 'secondary'} size="sm" icon={icReady ? 'circle-check-big' : 'list-checks'} disabled={!icReady}>{icReady ? 'Send to IC' : `${open.length} open`}</Button>
              <Button variant="ghost" size="sm" icon="file-text">Export memo</Button>
            </div>
          </div>
          {/* risk register */}
          {open.length > 0 && (
            <div className="card-flat" style={{ padding: '12px 14px', marginBottom: 14, background: 'var(--surface-1)' }}>
              <div className="t-eyebrow" style={{ marginBottom: 9, display: 'flex', alignItems: 'center', gap: 6 }}><Icon name="alert-triangle" size={12} color="var(--warning)" />Risk register · {open.length} open</div>
              <div style={{ display: 'flex', flexDirection: 'column', gap: 6 }}>
                {open.map(a => {
                  const st = agents[a.id]; const sev = DD_SEV[st.severity] || DD_SEV.Low;
                  return (
                    <button key={a.id} className="clickable" onClick={() => setOpenAgent(a.id)} style={{ display: 'flex', alignItems: 'center', gap: 10, padding: '8px 10px', borderRadius: 9, background: 'var(--surface-2)', border: '1px solid var(--border)', textAlign: 'left' }}>
                      <span style={{ fontSize: 9, fontWeight: 700, letterSpacing: '0.04em', color: '#fff', background: sev.c, borderRadius: 5, padding: '2px 6px', flex: 'none' }}>{(st.severity || 'Low').toUpperCase()}</span>
                      <span style={{ flex: 1, minWidth: 0, fontSize: 12, fontWeight: 500, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{st.headline}</span>
                      <span style={{ fontSize: 10.5, color: 'var(--fg-5)', flex: 'none' }}>{a.name}</span>
                      <Icon name="chevron-right" size={14} color="var(--fg-5)" />
                    </button>
                  );
                })}
              </div>
            </div>
          )}
          {/* agent grid */}
          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fill,minmax(250px,1fr))', gap: 10 }}>
            {DD_AGENTS.map(a => {
              const st = (agents[a.id] || {}); const s = DD_STATUS[st.status] || DD_STATUS.pending; const sev = st.severity && DD_SEV[st.severity];
              return (
                <button key={a.id} className="clickable card-flat" onClick={() => setOpenAgent(a.id)} style={{ textAlign: 'left', padding: '13px 14px', background: 'var(--surface-1)', borderLeft: `2px solid ${TONE[s.tone].c}` }}>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 10 }}>
                    <span style={{ width: 30, height: 30, borderRadius: 8, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--surface-2)', border: '1px solid var(--border)', color: 'var(--fg-3)' }}><Icon name={a.icon} size={15} /></span>
                    <div style={{ flex: 1, minWidth: 0 }}>
                      <div style={{ fontSize: 12.5, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{a.name}</div>
                      <div style={{ fontSize: 10, color: 'var(--fg-5)' }}>{a.who} · {st.checks} checks</div>
                    </div>
                    <span style={{ display: 'inline-flex', alignItems: 'center', gap: 4, fontSize: 9.5, fontWeight: 600, color: TONE[s.tone].c, flex: 'none' }}><Icon name={s.icon} size={11} />{s.label}</span>
                  </div>
                  <div style={{ fontSize: 11, color: 'var(--fg-4)', marginTop: 9, lineHeight: 1.4, display: '-webkit-box', WebkitLineClamp: 2, WebkitBoxOrient: 'vertical', overflow: 'hidden' }}>{st.headline}</div>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 8, marginTop: 10 }}>
                    <div style={{ flex: 1, height: 4, borderRadius: 999, background: 'var(--surface-3)', overflow: 'hidden' }}><div style={{ width: (st.confidence || 0) + '%', height: '100%', background: TONE[s.tone].c }} /></div>
                    <span style={{ fontSize: 10, color: 'var(--fg-5)', flex: 'none' }}>{st.confidence}%</span>
                    {sev && <span style={{ fontSize: 8.5, fontWeight: 700, color: '#fff', background: sev.c, borderRadius: 4, padding: '1px 5px', flex: 'none' }}>{st.severity.toUpperCase()}</span>}
                  </div>
                </button>
              );
            })}
          </div>
          <div className="card" style={{ padding: '13px 15px', marginTop: 14, display: 'flex', alignItems: 'center', gap: 11, background: 'var(--gold-soft)', borderColor: 'var(--gold-line)' }}>
            <EarnCoin size={24} />
            <div style={{ flex: 1, fontSize: 12, color: 'var(--fg-2)', lineHeight: 1.45 }}><b style={{ color: 'var(--gold-1)' }}>Earn:</b> {open.length ? `${verdict.label} — ${open.length} of ${total} workstreams need your call. Open the risk register and I'll resolve each with you.` : `All ${total} workstreams clear at ${avgConf}% avg confidence. This deal is IC-ready.`}</div>
          </div>
        </Panel>
      ) : tab === 'workflows' ? (
        <WorkflowsBoard deal={deal} wf={wfActive} stream={wfStream} setStream={setWfStream} autos={autos} setAutos={setAutos} onRunItem={(it) => runItem('wf-' + wfStream, setWfActive, it)} onRun={onRun} mobile={mobile} />
      ) : tab === 'compliance' ? (
        <ComplianceCenter items={co} onRunItem={(it) => runItem('co', setCo, it)} onRun={onRun} mobile={mobile} />
      ) : (
        <IRCenter items={ir} lps={IR_LPS} onRunItem={(it) => runItem('ir', setIr, it)} onRun={onRun} mobile={mobile} />
      )}

      {/* diligence detail drawer */}
      {ag && (
        <>
          <div onClick={() => setOpenAgent(null)} style={{ position: 'fixed', inset: 0, background: 'rgba(3,6,12,0.64)', backdropFilter: 'blur(3px)', zIndex: 60 }} />
          <div className="fade-in" style={{ position: 'fixed', top: 0, right: 0, bottom: 0, width: 440, maxWidth: '94vw', background: 'var(--bg-2)', borderLeft: '1px solid var(--border-strong)', zIndex: 61, overflowY: 'auto', boxShadow: '-30px 0 80px -30px rgba(0,0,0,0.7)' }}>
            <div style={{ padding: '18px 20px', borderBottom: '1px solid var(--border)', display: 'flex', alignItems: 'center', gap: 12 }}>
              <span style={{ width: 42, height: 42, borderRadius: 11, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--surface-2)', border: '1px solid var(--border)', color: 'var(--fg-2)' }}><Icon name={agMeta.icon} size={20} /></span>
              <div style={{ flex: 1, minWidth: 0 }}>
                <div style={{ fontSize: 15.5, fontWeight: 600 }}>{agMeta.name}</div>
                <div style={{ fontSize: 11.5, color: 'var(--fg-4)' }}>Run by {agMeta.who} · {deal.name}</div>
              </div>
              <button className="clickable" onClick={() => setOpenAgent(null)} style={{ width: 30, height: 30, borderRadius: 8, color: 'var(--fg-4)', display: 'flex', alignItems: 'center', justifyContent: 'center' }}><Icon name="x" size={17} /></button>
            </div>
            <div style={{ padding: 20, display: 'flex', flexDirection: 'column', gap: 16 }}>
              <div style={{ display: 'flex', alignItems: 'center', gap: 8, flexWrap: 'wrap' }}>
                <span style={{ display: 'inline-flex', alignItems: 'center', gap: 7, fontSize: 12, fontWeight: 600, color: TONE[DD_STATUS[ag.status].tone].c, background: TONE[DD_STATUS[ag.status].tone].bg, border: `1px solid ${TONE[DD_STATUS[ag.status].tone].b}`, borderRadius: 999, padding: '4px 11px' }}>
                  <Icon name={DD_STATUS[ag.status].icon} size={13} />{DD_STATUS[ag.status].label}
                </span>
                {ag.severity && ag.status !== 'clear' && <span style={{ fontSize: 9.5, fontWeight: 700, color: '#fff', background: (DD_SEV[ag.severity] || DD_SEV.Low).c, borderRadius: 5, padding: '3px 8px' }}>{ag.severity.toUpperCase()} SEVERITY</span>}
              </div>
              {/* results: confidence + checks */}
              <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: 10 }}>
                <div className="card-flat" style={{ padding: '11px 13px', background: 'var(--surface-1)' }}>
                  <div style={{ fontSize: 10, color: 'var(--fg-5)' }}>Confidence</div>
                  <div className="t-metric" style={{ fontSize: 18, marginTop: 4, color: TONE[DD_STATUS[ag.status].tone].c }}>{ag.confidence}%</div>
                </div>
                <div className="card-flat" style={{ padding: '11px 13px', background: 'var(--surface-1)' }}>
                  <div style={{ fontSize: 10, color: 'var(--fg-5)' }}>Checks run</div>
                  <div className="t-metric" style={{ fontSize: 18, marginTop: 4 }}>{ag.checks}</div>
                </div>
              </div>
              <div>
                <div style={{ fontSize: 14.5, fontWeight: 600 }}>{ag.headline}</div>
                <div style={{ fontSize: 12.5, color: 'var(--fg-3)', marginTop: 8, lineHeight: 1.6 }}>{ag.detail}</div>
              </div>
              {/* evidence reviewed — the results */}
              <div>
                <div className="t-eyebrow" style={{ marginBottom: 8, display: 'flex', alignItems: 'center', gap: 6 }}><Icon name="paperclip" size={12} color="var(--fg-4)" />Evidence reviewed · {(ag.evidence || []).length}</div>
                <div style={{ display: 'flex', flexDirection: 'column', gap: 5 }}>
                  {(ag.evidence || []).map((e, i) => (
                    <div key={i} style={{ display: 'flex', alignItems: 'center', gap: 9, padding: '8px 11px', borderRadius: 9, background: 'var(--surface-1)', border: '1px solid var(--border-faint)' }}>
                      <Icon name="file-check" size={14} color="var(--success)" />
                      <span style={{ flex: 1, fontSize: 12, color: 'var(--fg-2)' }}>{e}</span>
                      <Icon name="check" size={13} color="var(--fg-5)" />
                    </div>
                  ))}
                </div>
              </div>
              {ag.status !== 'clear' ? (
                <div style={{ padding: '14px 15px', borderRadius: 13, background: 'var(--gold-soft)', border: '1px solid var(--gold-line)' }}>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 9, marginBottom: 8 }}><EarnCoin size={24} /><span style={{ fontSize: 12.5, fontWeight: 600, color: 'var(--gold-1)' }}>Earn's resolution</span></div>
                  <div style={{ fontSize: 12, color: 'var(--fg-2)', lineHeight: 1.5, marginBottom: 11 }}>{ag.action} — I'll prepare it with {agMeta.who} and bring it back for your sign-off.</div>
                  <Button variant="gold" size="sm" icon="sparkles" style={{ width: '100%' }} onClick={() => { clearItem(openAgent); setOpenAgent(null); }}>{ag.action} with Earn</Button>
                </div>
              ) : (
                <div style={{ display: 'flex', alignItems: 'center', gap: 9, padding: '14px 15px', borderRadius: 13, background: 'var(--success-soft)', border: '1px solid var(--success-line)', color: 'var(--success)', fontSize: 13, fontWeight: 600 }}><Icon name="check-circle-2" size={17} />Cleared · logged to Chain of Trust</div>
              )}
            </div>
          </div>
        </>
      )}
    </div>
  );
}
