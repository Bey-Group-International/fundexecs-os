/* ---- shared: a specialist's mark (Earn coin or Lucide tile) ---- */
function SpecialistMark({ member, size = 36, tone = 'neutral' }) {
  if (member && member.icon === 'coin') return <EarnCoin size={size} />;
  const t = TONE[tone] || TONE.neutral;
  return (
    <span style={{ width: size, height: size, borderRadius: size * 0.28, flex: 'none', display: 'inline-flex',
      alignItems: 'center', justifyContent: 'center', background: t.bg, color: t.c, border: `1px solid ${t.b}` }}>
      <Icon name={(member && member.icon) || 'user'} size={Math.round(size * 0.46)} />
    </span>
  );
}
