function RadioRow({ opts, value, onChange }) {
  return (
    <div style={{ display: 'flex', flexWrap: 'wrap', gap: 8 }}>
      {opts.map(o => <Chip key={o} label={o} selected={value === o} onClick={() => onChange(o)} />)}
    </div>
  );
}

function MaterialBuilder({ item, mobile, onBack, onComplete }) {
  const cfg = MATERIAL_BUILD[item.id] || { intro: '', decisions: [], rec: {}, recText: '' };
  const meta = MAT_META[item.id] || {};
  const [d, setD] = useState(() => ({ ...cfg.rec }));
  const [applied, setApplied] = useState(false);
  const [phase, setPhase] = useState('edit'); // edit | building | done
  const set = (k, v) => setD(p => ({ ...p, [k]: v }));
  const toggle = (k, v) => setD(p => { const cur = p[k] || []; return { ...p, [k]: cur.includes(v) ? cur.filter(x => x !== v) : [...cur, v] }; });
  const applyRec = () => { setD({ ...cfg.rec }); setApplied(true); };

  const steps = ['Pull your fund story & data', `Draft your ${item.name.toLowerCase()}`, 'Format to institutional standard', `Place in ${meta.folder || 'the data room'}`];
  const [n, setN] = useState(0);
  useEffect(() => {
    if (phase !== 'building') return;
    let i = 0; const t = setInterval(() => { i += 1; setN(i); if (i >= steps.length) { clearInterval(t); setTimeout(() => setPhase('done'), 450); } }, 600);
    return () => clearInterval(t);
  }, [phase]);

  const rows = cfg.decisions.map(dec => [dec.label, Array.isArray(d[dec.key]) ? (d[dec.key].length ? d[dec.key].join(', ') : 'None') : d[dec.key]]);

  /* DONE — results laid out */
  if (phase === 'done') {
    return (
      <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', padding: '20px 0' }}>
        <div style={{ width: '100%', maxWidth: 560 }}>
          <div style={{ textAlign: 'center' }}>
            <div style={{ width: 56, height: 56, borderRadius: '50%', margin: '0 auto 14px', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--success-soft)', border: '1px solid var(--success-line)' }}><Icon name="check" size={28} color="var(--success)" /></div>
            <div className="t-h2" style={{ fontSize: 20 }}>{item.name} — ready</div>
            <p style={{ fontSize: 13, color: 'var(--fg-3)', marginTop: 7 }}>Built to your spec and placed in {meta.folder}. Review it, then add more — or go back and adjust.</p>
          </div>
          <div style={{ borderRadius: 14, border: '1px solid var(--border)', overflow: 'hidden', marginTop: 18 }}>
            <div style={{ display: 'flex', alignItems: 'center', gap: 9, padding: '11px 15px', background: 'var(--surface-1)', borderBottom: '1px solid var(--border)' }}>
              <Icon name={meta.icon || 'file-text'} size={15} color="var(--gold-1)" />
              <span style={{ fontSize: 12.5, fontWeight: 600 }}>{item.name}</span>
              <span style={{ flex: 1 }} />
              <span style={{ display: 'inline-flex', alignItems: 'center', gap: 5, fontSize: 10.5, color: 'var(--success)', fontWeight: 600 }}><Icon name="folder-lock" size={12} />In {meta.folder}</span>
            </div>
            {rows.map(([k, v], i) => (
              <div key={k} style={{ display: 'flex', gap: 14, padding: '10px 15px', background: i % 2 ? 'transparent' : 'var(--surface-1)', borderTop: i ? '1px solid var(--border-faint)' : 'none' }}>
                <span style={{ width: 130, flex: 'none', fontSize: 12, color: 'var(--fg-4)' }}>{k}</span>
                <span style={{ fontSize: 13, fontWeight: 500, color: 'var(--fg-1)' }}>{v}</span>
              </div>
            ))}
          </div>
          <div style={{ display: 'flex', gap: 10, justifyContent: 'space-between', alignItems: 'center', flexWrap: 'wrap', marginTop: 20 }}>
            <Button variant="ghost" icon="pencil" onClick={() => { setPhase('edit'); setN(0); }}>Go back &amp; edit</Button>
            <Button variant="gold" iconRight="arrow-right" onClick={onComplete}>Add to room &amp; continue</Button>
          </div>
        </div>
      </div>
    );
  }

  /* BUILDING */
  if (phase === 'building') {
    return (
      <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', padding: '24px 0' }}>
        <div style={{ width: '100%', maxWidth: 540 }}>
          <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', textAlign: 'center', marginBottom: 18 }}>
            <div style={{ position: 'relative', marginBottom: 13 }}>
              <div className="pulse-dot" style={{ position: 'absolute', inset: -10, borderRadius: '50%', background: 'radial-gradient(circle, rgba(247,201,72,0.5), transparent 70%)', filter: 'blur(8px)' }} />
              <div style={{ position: 'relative' }}><EarnCoin size={52} /></div>
            </div>
            <div className="t-h2" style={{ fontSize: 19 }}>Building {item.name}…</div>
            <p style={{ fontSize: 12.5, color: 'var(--fg-3)', marginTop: 6 }}>Drafting from your fund story to the spec you set.</p>
          </div>
          <ProgressBar value={Math.round((n / steps.length) * 100)} gradient="linear-gradient(90deg,#F7C948,#E5A823)" height={6} />
          <div className="card" style={{ marginTop: 14, padding: 12, display: 'flex', flexDirection: 'column', gap: 2 }}>
            {steps.map((s, i) => i <= n && (
              <div key={i} className="fade-in" style={{ display: 'flex', alignItems: 'center', gap: 11, padding: '9px 8px' }}>
                <span style={{ width: 22, height: 22, borderRadius: 999, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: i < n ? 'var(--success-soft)' : 'var(--surface-2)', border: `1px solid ${i < n ? 'var(--success-line)' : 'var(--border)'}`, color: i < n ? 'var(--success)' : 'var(--fg-4)' }}>
                  {i < n ? <Icon name="check" size={12} /> : <Icon name="loader" size={12} className="spin" />}</span>
                <span style={{ fontSize: 13, color: i < n ? 'var(--fg-2)' : 'var(--fg-1)' }}>{s}</span>
              </div>
            ))}
          </div>
        </div>
      </div>
    );
  }

  /* EDIT — copiloted decisions + Earn */
  return (
    <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
      <div style={{ display: 'flex', alignItems: 'center', gap: 12 }}>
        <button className="clickable" onClick={onBack} style={{ display: 'inline-flex', alignItems: 'center', gap: 6, fontSize: 12.5, color: 'var(--fg-3)', padding: '7px 11px', borderRadius: 9, border: '1px solid var(--border)', background: 'var(--surface-1)' }}><Icon name="arrow-left" size={14} />Materials</button>
        <div style={{ flex: 1, minWidth: 0 }}>
          <div className="t-h2" style={{ fontSize: 18 }}>{item.name}</div>
          <div style={{ fontSize: 12, color: 'var(--fg-4)' }}>You shape it, Earn drafts it · lands in {meta.folder}</div>
        </div>
        <Badge tone="azure" dot>Copiloted</Badge>
      </div>

      <div style={{ display: 'grid', gridTemplateColumns: mobile ? '1fr' : 'minmax(0,1fr) 300px', gap: 16, alignItems: 'start' }}>
        <div className="card" style={{ padding: 22 }}>
          <p style={{ fontSize: 12.5, color: 'var(--fg-4)', margin: '0 0 18px', lineHeight: 1.55 }}>{cfg.intro}</p>
          <div style={{ display: 'flex', flexDirection: 'column', gap: 20 }}>
            {cfg.decisions.map(dec => (
              <div key={dec.key}>
                <div className="t-eyebrow" style={{ marginBottom: 9 }}>{dec.label}{dec.kind === 'multi' && <span style={{ color: 'var(--fg-5)', fontWeight: 400, textTransform: 'none', letterSpacing: 0 }}> · pick any</span>}</div>
                {dec.kind === 'radio'
                  ? <RadioRow opts={dec.opts} value={d[dec.key]} onChange={v => set(dec.key, v)} />
                  : <div style={{ display: 'flex', flexWrap: 'wrap', gap: 8 }}>{dec.opts.map(o => <Chip key={o} label={o} selected={(d[dec.key] || []).includes(o)} onClick={() => toggle(dec.key, o)} />)}</div>}
              </div>
            ))}
          </div>
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginTop: 24, gap: 10, flexWrap: 'wrap' }}>
            <Button variant="ghost" icon="arrow-left" onClick={onBack}>Cancel</Button>
            <Button variant="gold" iconRight="sparkles" onClick={() => { setN(0); setPhase('building'); }}>Build &amp; add to room</Button>
          </div>
        </div>

        <div className="card" style={{ padding: 17, alignSelf: 'flex-start' }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: 10, marginBottom: 11 }}>
            <div style={{ position: 'relative', flex: 'none' }}><EarnCoin size={32} /><span className="pulse-dot" style={{ position: 'absolute', right: -1, bottom: -1, width: 9, height: 9, borderRadius: 999, background: 'var(--success)', border: '2px solid var(--bg-2)' }} /></div>
            <div><div style={{ fontSize: 13, fontWeight: 600 }}>Earn</div><div style={{ fontSize: 10.5, color: 'var(--fg-4)' }}>{meta.cat} copilot</div></div>
          </div>
          <div className="t-eyebrow" style={{ marginBottom: 7, color: 'var(--gold-1)' }}>Earn recommends</div>
          <div style={{ fontSize: 12.5, color: 'var(--fg-2)', lineHeight: 1.6 }}>{cfg.recText}</div>
          <Button variant={applied ? 'secondary' : 'gold'} size="sm" icon={applied ? 'check' : 'sparkles'} style={{ width: '100%', marginTop: 14 }} onClick={applyRec}>{applied ? 'Recommendation applied' : "Apply Earn's recommendation"}</Button>
          <div style={{ display: 'flex', alignItems: 'center', gap: 6, marginTop: 12, fontSize: 11, color: 'var(--fg-5)' }}><Icon name="hand" size={12} />You're in control — change anything.</div>
        </div>
      </div>
    </div>
  );
}

  const [name, setName] = useState(prospect.name);
  const [pfirm, setPfirm] = useState(prospect.firm);

  const [nda, setNda] = useState(false);

          </div>
        )}
      </div>
    </>
  );
}

  /* every shareable document = formation docs built + materials ready + adopted policies */
  const roomDocs = [
    ...formDocsBuilt.map(id => ({ key: 'f-' + id, name: FORM_DOC_META[id].name, folder: FORM_DOC_META[id].folder, src: 'Formation', icon: 'file-text' })),
    ...MAT_DOCS.filter(id => matById(id).stage === 'Ready').map(id => ({ key: 'm-' + id, name: matById(id).name, folder: (MAT_META[id] || {}).folder, src: 'Built here', icon: (MAT_META[id] || {}).icon || 'file-text' })),
    ...govPolicies.map(p => ({ key: 'g-' + p.id, name: p.name, folder: 'Governance', src: 'Governance', icon: 'scale' })),
  ];
  const token = () => Math.random().toString(36).slice(2, 6) + '-' + Math.random().toString(36).slice(2, 6);
  function generateLink(name) {
    setLinks(p => p[name] ? p : { ...p, [name]: { token: token(), vetting: 'Accredited + NDA', viewers: [] } });
    setActivity(a => [{ who: 'You', act: `generated a secure link for ${name}`, t: 'Just now', icon: 'link' }, ...a]);
  }
  function verifyViewer(name, who) {
    setLinks(p => ({ ...p, [name]: { ...p[name], viewers: [...(p[name].viewers || []), { ...who, t: 'Just now' }] } }));
    setActivity(a => [{ who: who.name, act: `was vetted & opened ${name}`, t: 'Just now', icon: 'shield-check' }, ...a]);
    setProspIdx(i => i + 1);
    setGate(null);
  }

  function buildMaterial(id) {
    const m = matById(id); const meta = MAT_META[id];
    markReady && markReady(id);
    onMaterialBuilt && onMaterialBuilt(m.name);
    setActivity(a => [{ who: 'You', act: `added ${m.name} to ${meta.folder}`, t: 'Just now', icon: 'file-plus' }, ...a]);
    setOpenMat(null);
  }
  function inviteLp() {
    const lp = DR_INVITEES[invIdx % DR_INVITEES.length];
    onRun({ id: 'dr-invite-' + lp.id + '-' + invIdx, hub: 'build', xp: 10, title: `Invite ${lp.name} to the data room`,
      apply: () => { setAccess(a => [...a, { id: lp.id + invIdx, name: lp.name, type: lp.type, status: 'Invited', tone: 'neutral' }]); setActivity(a => [{ who: 'You', act: `invited ${lp.name}`, t: 'Just now', icon: 'user-plus' }, ...a]); setInvIdx(i => i + 1); },
      run: { title: `Invite ${lp.name}`, steps: ['Prepare a scoped, watermarked access link', 'Draft the invitation note', 'Set view + download permissions', 'Prepare for your approval'],
        draftTitle: `Invite — ${lp.name}`, draft: `Earn drafted a secure, tracked invitation for ${lp.name} (${lp.type}). They'll get scoped access and every view is logged. Approve to send.` } });
  }

  if (openMat) {
    const item = mats.find(m => m.id === openMat) || {};
    return <MaterialBuilder key={openMat} item={item} mobile={mobile} onBack={() => setOpenMat(null)} onComplete={() => buildMaterial(openMat)} />;
  }

  return (
    <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
      {/* readiness header */}
      <div className="card" style={{ padding: '18px 20px' }}>
        <div style={{ display: 'flex', alignItems: 'center', gap: 13 }}>
          <span style={{ width: 44, height: 44, borderRadius: 12, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--accent-soft)', border: '1px solid var(--accent-line)', color: 'var(--accent)' }}><Icon name="folder-lock" size={22} /></span>
          <div style={{ flex: 1, minWidth: 0 }}>
            <div className="t-h2" style={{ fontSize: 19 }}>Materials &amp; data room</div>
            <div style={{ fontSize: 12.5, color: 'var(--fg-3)', marginTop: 2 }}>Everything an LP asks for — generated by the team, served from one secure room.</div>
          </div>
          <div style={{ textAlign: 'right', flex: 'none' }}>
            <div className="t-metric" style={{ fontSize: 22, color: 'var(--gold-1)' }}>{readyCount}/{MAT_DOCS.length}</div>
            <div style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>LP-ready</div>
          </div>
        </div>
        <div style={{ marginTop: 14 }}><ProgressBar value={roomReady} gradient="linear-gradient(90deg,#F7C948,#E5A823)" height={6} /></div>
      </div>

      <div style={{ overflowX: 'auto', overflowY: 'hidden', paddingBottom: 2 }}>
        <SegTabs active={view} onChange={setView} tabs={[{ id: 'materials', label: 'Investor materials', icon: 'files' }, { id: 'room', label: 'The data room', icon: 'folder-lock' }]} />
      </div>

      {view === 'materials' ? (
        <Panel title="Investor materials" eyebrow="The documents formation doesn't build · drafted from your fund story" icon="files"
          action={<Button variant="ghost" size="sm" icon="sparkles" onClick={() => onRun(ASK_EARN)}>Generate all</Button>}>
          <div style={{ display: 'flex', alignItems: 'flex-start', gap: 9, marginBottom: 12, padding: '10px 13px', borderRadius: 11, background: 'var(--surface-1)', border: '1px solid var(--border-faint)' }}>
            <Icon name="info" size={14} color="var(--gold-1)" style={{ marginTop: 1, flex: 'none' }} />
            <span style={{ fontSize: 11.5, color: 'var(--fg-4)', lineHeight: 1.5 }}>Your legal set — LPA, PPM, subscription docs, Form D — is built in <b style={{ color: 'var(--fg-2)' }}>Formation</b> and flows into the room automatically. Build the investor-facing materials here.</span>
          </div>
          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit,minmax(240px,1fr))', gap: 10 }}>
            {MAT_DOCS.map(id => {
              const m = matById(id); const meta = MAT_META[id]; const ready = m.stage === 'Ready';
              return (
                <div key={id} className="card-flat" style={{ display: 'flex', flexDirection: 'column', padding: '14px 15px', background: 'var(--surface-1)' }}>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 10 }}>
                    <span style={{ width: 34, height: 34, borderRadius: 9, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: ready ? 'var(--success-soft)' : 'var(--surface-2)', border: `1px solid ${ready ? 'var(--success-line)' : 'var(--border)'}`, color: ready ? 'var(--success)' : 'var(--fg-3)' }}><Icon name={meta.icon} size={17} /></span>
                    <div style={{ flex: 1, minWidth: 0 }}>
                      <div style={{ fontSize: 13, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{m.name}</div>
                      <div style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>{meta.cat} · {meta.fmt}</div>
                    </div>
                  </div>
                  <div style={{ display: 'flex', alignItems: 'center', gap: 8, marginTop: 13 }}>
                    <Badge tone={MAT_TONE[m.stage]} style={{ fontSize: 9.5, padding: '2px 7px' }}>{m.stage}</Badge>
                    <span style={{ flex: 1 }} />
                    {ready
                      ? <Button variant="ghost" size="sm" icon="eye">Open</Button>
                      : <Button variant="secondary" size="sm" icon="sparkles" onClick={() => setOpenMat(id)}>{MAT_NEXT[m.stage]}</Button>}
                  </div>
                </div>
              );
            })}
          </div>
        </Panel>
      ) : (
        <div style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
          {/* live room banner */}
          <div className="card" style={{ padding: '15px 18px', display: 'flex', alignItems: 'center', gap: 13, flexWrap: 'wrap', background: 'linear-gradient(100deg, rgba(247,201,72,0.10), transparent 60%)', borderColor: 'var(--gold-line)' }}>
            <span className="pulse-dot" style={{ width: 9, height: 9, borderRadius: 999, background: 'var(--success)', flex: 'none' }} />
            <div style={{ flex: 1, minWidth: 0 }}>
              <div style={{ fontSize: 13.5, fontWeight: 600 }}>Your data room is live</div>
              <div style={{ fontSize: 11.5, color: 'var(--fg-4)' }}>{access.length} LPs with access · {readyCount + formDocsBuilt.length + govPolicies.length} documents · every view tracked to your Chain of Trust</div>
            </div>
            <Button variant="gold" size="sm" icon="user-plus" onClick={inviteLp}>Invite an LP</Button>
            <Button variant="outline" size="sm" icon="link">Copy link</Button>
          </div>

          <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit,minmax(280px,1fr))', gap: 14, alignItems: 'start' }}>
            {/* documents + per-document secure links */}
            <Panel title="Documents" eyebrow="Generate a vetted, unique link per document" icon="folder-lock">
              {roomDocs.length === 0 && <div style={{ fontSize: 12, color: 'var(--fg-5)', padding: '8px 2px' }}>No documents yet — finalize materials or finish Formation.</div>}
              <div style={{ display: 'flex', flexDirection: 'column', gap: 8 }}>
                {roomDocs.map(doc => {
                  const link = links[doc.name];
                  return (
                    <div key={doc.key} className="card-flat" style={{ padding: '11px 13px', background: 'var(--surface-1)' }}>
                      <div style={{ display: 'flex', alignItems: 'center', gap: 11 }}>
                        <span style={{ width: 30, height: 30, borderRadius: 8, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--surface-2)', border: '1px solid var(--border)', color: 'var(--fg-3)' }}><Icon name={doc.icon} size={15} /></span>
                        <div style={{ flex: 1, minWidth: 0 }}>
                          <div style={{ fontSize: 12.5, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{doc.name}</div>
                          <div style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>{doc.folder} · {doc.src}</div>
                        </div>
                        {link
                          ? <span style={{ display: 'inline-flex', alignItems: 'center', gap: 5, fontSize: 10.5, fontWeight: 600, color: 'var(--success)', flex: 'none' }}><Icon name="check-circle-2" size={12} />Link live</span>
                          : <Button variant="secondary" size="sm" icon="link" style={{ flex: 'none' }} onClick={() => generateLink(doc.name)}>Generate link</Button>}
                      </div>
                      {link && (
                        <div className="fade-in" style={{ marginTop: 10, paddingTop: 10, borderTop: '1px solid var(--border-faint)' }}>
                          <div style={{ display: 'flex', alignItems: 'center', gap: 8, flexWrap: 'wrap' }}>
                            <span className="t-mono" style={{ fontSize: 11, color: 'var(--accent)', background: 'var(--accent-soft)', border: '1px solid var(--accent-line)', borderRadius: 7, padding: '3px 8px' }}>fundexecs.io/dr/{link.token}</span>
                            <span style={{ display: 'inline-flex', alignItems: 'center', gap: 4, fontSize: 10.5, color: 'var(--fg-4)' }}><Icon name="shield-check" size={12} color="var(--gold-1)" />Vets: {link.vetting}</span>
                            <span style={{ flex: 1 }} />
                            <Button variant="ghost" size="sm" icon="external-link" onClick={() => setGate({ docName: doc.name })}>Open as recipient</Button>
                          </div>
                          {/* who opened it */}
                          <div style={{ marginTop: 9 }}>
                            <div style={{ fontSize: 10.5, color: 'var(--fg-5)', marginBottom: 6 }}>{link.viewers.length ? `${link.viewers.length} ${link.viewers.length === 1 ? 'person has' : 'people have'} opened this link` : 'No one has opened this link yet'}</div>
                            {link.viewers.map((v, i) => (
                              <div key={i} style={{ display: 'flex', alignItems: 'center', gap: 9, padding: '6px 0' }}>
                                <Avatar name={v.name} size={24} tone="azure" />
                                <div style={{ flex: 1, minWidth: 0 }}>
                                  <div style={{ fontSize: 12, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{v.name} <span style={{ fontWeight: 400, color: 'var(--fg-5)' }}>· {v.firm}</span></div>
                                </div>
                                <span style={{ display: 'inline-flex', alignItems: 'center', gap: 4, fontSize: 10, fontWeight: 600, color: 'var(--success)', flex: 'none' }}><Icon name="shield-check" size={11} />Vetted</span>
                                <span style={{ fontSize: 10.5, color: 'var(--fg-5)', flex: 'none' }}>{v.t}</span>
                              </div>
                            ))}
                          </div>
                        </div>
                      )}
                    </div>
                  );
                })}
              </div>
            </Panel>

            {/* access */}
            <Panel title="Who has access" eyebrow={`${access.length} LPs · scoped & tracked`} icon="users">
              <div style={{ display: 'flex', flexDirection: 'column', gap: 6 }}>
                {access.map(lp => (
                  <div key={lp.id} className="card-flat" style={{ display: 'flex', alignItems: 'center', gap: 11, padding: '10px 13px', background: 'var(--surface-1)' }}>
                    <Avatar name={lp.name} size={30} tone={lp.tone === 'success' ? 'gold' : 'azure'} />
                    <div style={{ flex: 1, minWidth: 0 }}>
                      <div style={{ fontSize: 12.5, fontWeight: 600, whiteSpace: 'nowrap', overflow: 'hidden', textOverflow: 'ellipsis' }}>{lp.name}</div>
                      <div style={{ fontSize: 10.5, color: 'var(--fg-5)' }}>{lp.type}</div>
                    </div>
                    <span style={{ display: 'inline-flex', alignItems: 'center', gap: 5, fontSize: 10.5, fontWeight: 600, color: TONE[lp.tone].c, flex: 'none' }}>
                      {lp.live && <span className="pulse-dot" style={{ width: 6, height: 6, borderRadius: 999, background: 'var(--success)' }} />}{lp.status}
                    </span>
                  </div>
                ))}
              </div>
            </Panel>
          </div>

          {/* activity */}
          <Panel title="Room activity" eyebrow="Every interaction, on the record" icon="activity">
            <div style={{ display: 'flex', flexDirection: 'column' }}>
              {activity.map((a, i) => (
                <div key={i} style={{ display: 'flex', alignItems: 'center', gap: 11, padding: '9px 2px', borderTop: i ? '1px solid var(--border-faint)' : 'none' }}>
                  <span style={{ width: 28, height: 28, borderRadius: 8, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--surface-2)', border: '1px solid var(--border)', color: 'var(--fg-3)' }}><Icon name={a.icon} size={14} /></span>
                  <div style={{ flex: 1, minWidth: 0, fontSize: 12.5, color: 'var(--fg-2)' }}><b style={{ color: 'var(--fg-1)' }}>{a.who}</b> {a.act}</div>
                  <span style={{ fontSize: 11, color: 'var(--fg-5)', flex: 'none' }}>{a.t}</span>
                </div>
              ))}
            </div>
          </Panel>
        </div>
      )}

      <div className="card" style={{ padding: '14px 16px', display: 'flex', alignItems: 'center', gap: 12, background: 'var(--gold-soft)', borderColor: 'var(--gold-line)' }}>
        <EarnCoin size={26} />
        <div style={{ flex: 1, fontSize: 12.5, color: 'var(--fg-2)', lineHeight: 1.5 }}><b style={{ color: 'var(--gold-1)' }}>Earn:</b> I draft every document from your fund story and keep the room current. Finalize a material and it lands in the right folder automatically.</div>
      </div>

      {gate && <VettingGate docName={gate.docName} firm={mandate.firm || 'the fund'} prospect={DR_PROSPECTS[prospIdx % DR_PROSPECTS.length]}
        onClose={() => setGate(null)} onVerify={(who) => verifyViewer(gate.docName, who)} />}
    </div>
  );
}
