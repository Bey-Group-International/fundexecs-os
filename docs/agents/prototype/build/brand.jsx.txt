/* generic copiloted brand builder (mirrors the materials / policy builders) */
function BrandBuilder({ item, mobile, onBack, onComplete }) {
  const cfg = BRAND_BUILD[item.id] || { intro: '', decisions: [], rec: {}, recText: '' };
  const [d, setD] = useState(() => ({ ...cfg.rec }));
  const [applied, setApplied] = useState(false);
  const [phase, setPhase] = useState('edit');
  const set = (k, v) => setD(p => ({ ...p, [k]: v }));
  const toggle = (k, v) => setD(p => { const c = p[k] || []; return { ...p, [k]: c.includes(v) ? c.filter(x => x !== v) : [...c, v] }; });
  const steps = ['Pull your fund story & profile', `Produce your ${item.name.toLowerCase()}`, 'Polish to institutional standard', 'Publish to your workspace'];
  const [n, setN] = useState(0);
  useEffect(() => { if (phase !== 'building') return; let i = 0; const t = setInterval(() => { i += 1; setN(i); if (i >= steps.length) { clearInterval(t); setTimeout(() => setPhase('done'), 420); } }, 580); return () => clearInterval(t); }, [phase]);
  const rows = cfg.decisions.map(dec => [dec.label, Array.isArray(d[dec.key]) ? (d[dec.key].join(', ') || 'None') : d[dec.key]]);

  if (phase === 'done') {
    return (
      <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', padding: '20px 0' }}>
        <div style={{ width: '100%', maxWidth: 540 }}>
          <div style={{ textAlign: 'center' }}>
            <div style={{ width: 54, height: 54, borderRadius: '50%', margin: '0 auto 13px', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--success-soft)', border: '1px solid var(--success-line)' }}><Icon name="check" size={26} color="var(--success)" /></div>
            <div className="t-h2" style={{ fontSize: 19 }}>{item.name} — live</div>
            <p style={{ fontSize: 12.5, color: 'var(--fg-3)', marginTop: 6 }}>Produced to your spec and published across your workspace.</p>
          </div>
          <div style={{ borderRadius: 13, border: '1px solid var(--border)', overflow: 'hidden', marginTop: 16 }}>
            {rows.map(([k, v], i) => (
              <div key={k} style={{ display: 'flex', gap: 14, padding: '10px 15px', background: i % 2 ? 'transparent' : 'var(--surface-1)', borderTop: i ? '1px solid var(--border-faint)' : 'none' }}>
                <span style={{ width: 120, flex: 'none', fontSize: 12, color: 'var(--fg-4)' }}>{k}</span>
                <span style={{ fontSize: 13, fontWeight: 500, color: 'var(--fg-1)' }}>{v}</span>
              </div>
            ))}
          </div>
          <div style={{ display: 'flex', gap: 10, justifyContent: 'space-between', marginTop: 18, flexWrap: 'wrap' }}>
            <Button variant="ghost" icon="pencil" onClick={() => { setPhase('edit'); setN(0); }}>Go back &amp; edit</Button>
            <Button variant="gold" iconRight="check" onClick={() => onComplete(d)}>Publish &amp; finish</Button>
          </div>
        </div>
      </div>
    );
  }
  if (phase === 'building') {
    return (
      <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', padding: '24px 0' }}>
        <div style={{ width: '100%', maxWidth: 520 }}>
          <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', textAlign: 'center', marginBottom: 16 }}>
            <div style={{ position: 'relative', marginBottom: 12 }}><div className="pulse-dot" style={{ position: 'absolute', inset: -10, borderRadius: '50%', background: 'radial-gradient(circle, rgba(247,201,72,0.5), transparent 70%)', filter: 'blur(8px)' }} /><div style={{ position: 'relative' }}><EarnCoin size={48} /></div></div>
            <div className="t-h2" style={{ fontSize: 18 }}>Producing {item.name}…</div>
          </div>
          <ProgressBar value={Math.round((n / steps.length) * 100)} gradient="linear-gradient(90deg,#F7C948,#E5A823)" height={6} />
          <div className="card" style={{ marginTop: 13, padding: 12, display: 'flex', flexDirection: 'column', gap: 2 }}>
            {steps.map((s, i) => i <= n && (
              <div key={i} className="fade-in" style={{ display: 'flex', alignItems: 'center', gap: 11, padding: '8px' }}>
                <span style={{ width: 22, height: 22, borderRadius: 999, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: i < n ? 'var(--success-soft)' : 'var(--surface-2)', border: `1px solid ${i < n ? 'var(--success-line)' : 'var(--border)'}`, color: i < n ? 'var(--success)' : 'var(--fg-4)' }}>{i < n ? <Icon name="check" size={12} /> : <Icon name="loader" size={12} className="spin" />}</span>
                <span style={{ fontSize: 13, color: i < n ? 'var(--fg-2)' : 'var(--fg-1)' }}>{s}</span>
              </div>
            ))}
          </div>
        </div>
      </div>
    );
  }
  return (
    <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
      <div style={{ display: 'flex', alignItems: 'center', gap: 12 }}>
        <button className="clickable" onClick={onBack} style={{ display: 'inline-flex', alignItems: 'center', gap: 6, fontSize: 12.5, color: 'var(--fg-3)', padding: '7px 11px', borderRadius: 9, border: '1px solid var(--border)', background: 'var(--surface-1)' }}><Icon name="arrow-left" size={14} />Back</button>
        <div style={{ flex: 1, minWidth: 0 }}><div className="t-h2" style={{ fontSize: 18 }}>{item.name}</div><div style={{ fontSize: 12, color: 'var(--fg-4)' }}>You set the direction, Earn produces it</div></div>
        <Badge tone="azure" dot>Copiloted</Badge>
      </div>
      <div style={{ display: 'grid', gridTemplateColumns: mobile ? '1fr' : 'minmax(0,1fr) 300px', gap: 16, alignItems: 'start' }}>
        <div className="card" style={{ padding: 22 }}>
          <p style={{ fontSize: 12.5, color: 'var(--fg-4)', margin: '0 0 18px', lineHeight: 1.55 }}>{cfg.intro}</p>
          <div style={{ display: 'flex', flexDirection: 'column', gap: 20 }}>
            {cfg.decisions.map(dec => (
              <div key={dec.key}>
                <div className="t-eyebrow" style={{ marginBottom: 9 }}>{dec.label}{dec.kind === 'multi' && <span style={{ color: 'var(--fg-5)', fontWeight: 400, textTransform: 'none', letterSpacing: 0 }}> · pick any</span>}</div>
                <div style={{ display: 'flex', flexWrap: 'wrap', gap: 8 }}>
                  {dec.opts.map(o => <Chip key={o} label={o} selected={dec.kind === 'multi' ? (d[dec.key] || []).includes(o) : d[dec.key] === o} onClick={() => dec.kind === 'multi' ? toggle(dec.key, o) : set(dec.key, o)} />)}
                </div>
              </div>
            ))}
          </div>
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginTop: 24, gap: 10, flexWrap: 'wrap' }}>
            <Button variant="ghost" icon="arrow-left" onClick={onBack}>Cancel</Button>
            <Button variant="gold" iconRight="sparkles" onClick={() => { setN(0); setPhase('building'); }}>Produce &amp; publish</Button>
          </div>
        </div>
        <div className="card" style={{ padding: 17, alignSelf: 'flex-start' }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: 10, marginBottom: 11 }}><div style={{ position: 'relative', flex: 'none' }}><EarnCoin size={32} /><span className="pulse-dot" style={{ position: 'absolute', right: -1, bottom: -1, width: 9, height: 9, borderRadius: 999, background: 'var(--success)', border: '2px solid var(--bg-2)' }} /></div><div><div style={{ fontSize: 13, fontWeight: 600 }}>Earn</div><div style={{ fontSize: 10.5, color: 'var(--fg-4)' }}>Brand copilot</div></div></div>
          <div className="t-eyebrow" style={{ marginBottom: 7, color: 'var(--gold-1)' }}>Earn recommends</div>
          <div style={{ fontSize: 12.5, color: 'var(--fg-2)', lineHeight: 1.6 }}>{cfg.recText}</div>
          <Button variant={applied ? 'secondary' : 'gold'} size="sm" icon={applied ? 'check' : 'sparkles'} style={{ width: '100%', marginTop: 14 }} onClick={() => { setD({ ...cfg.rec }); setApplied(true); }}>{applied ? 'Recommendation applied' : "Apply Earn's recommendation"}</Button>
          <div style={{ display: 'flex', alignItems: 'center', gap: 6, marginTop: 12, fontSize: 11, color: 'var(--fg-5)' }}><Icon name="hand" size={12} />You're in control — change anything.</div>
        </div>
      </div>
    </div>
  );
}

function BrandStat({ label, value, sub, tone = 'gold' }) {
  return (
    <div className="card-flat" style={{ padding: '12px 14px', background: 'var(--surface-1)' }}>
      <div style={{ fontSize: 11, color: 'var(--fg-4)' }}>{label}</div>
      <div className="t-metric" style={{ fontSize: 19, marginTop: 6, color: TONE[tone].c }}>{value}</div>
      {sub && <div style={{ fontSize: 10.5, color: 'var(--fg-5)', marginTop: 2 }}>{sub}</div>}
    </div>
  );
}

/* compose an actual bio paragraph from the inputs + posture */
function composeBio(d, principal, firm) {
  const first = (principal || 'Jordan Avery').split(' ')[0];
  const yrs = d.years || '10';
  const prior = d.prior || 'a decade operating in the sector';
  const win = d.win || 'three exits and a portfolio built on conviction';
  const edu = d.edu || '';
  const voiceOpen = {
    Operator: `${principal} is the Managing Partner of ${firm} — an operator-turned-investor who spent ${yrs} years building in the sector before backing it.`,
    Investor: `${principal} is the Managing Partner of ${firm}, an investor with ${yrs} years of disciplined sector focus and a track record of finding value others miss.`,
    Visionary: `${principal} is the Managing Partner of ${firm}, building the firm to back the operators and ideas the market is only now catching up to.`,
  }[d.voice || 'Operator'];
  const parts = [voiceOpen];
  if ((d.include || []).includes('Operating wins')) parts.push(`Before ${firm}, ${first} led ${prior.toLowerCase().startsWith('a ') ? prior : 'work at ' + prior} — earning the relationships and pattern-recognition that now drive the fund's edge.`);
  if ((d.include || []).includes('Track record')) parts.push(`That work produced ${win}.`);
  if ((d.include || []).includes('Thesis')) parts.push(`Today ${first} invests behind a focused thesis: backing overlooked operators with an unfair sourcing edge.`);
  if ((d.include || []).includes('Network')) parts.push(`${first}'s network across the sector turns into proprietary deal flow and warm capital.`);
  if ((d.include || []).includes('Education') && edu) parts.push(`${first} holds ${edu}.`);
  const full = parts.join(' ');
  if (d.length === 'Short') return parts.slice(0, 2).join(' ');
  if (d.length === 'Full') return full + ` ${first} writes and speaks regularly on the space, and sits at the center of the conversations that shape it.`;
  return full;
}

const TR_RECOGNITION_OPTS = ['Forbes 30 Under 30', '2 board seats', 'Ex-Head of Ops, sector incumbent', 'Published author', 'Industry award', 'Prior exit founder'];

function trAgg(deals) {
  const valid = deals.filter(d => d.company && parseFloat(d.multiple) > 0);
  const mults = valid.map(d => parseFloat(d.multiple));
  const realized = valid.filter(d => d.status === 'Realized').length;
  const blended = mults.length ? (mults.reduce((a, b) => a + b, 0) / mults.length) : 0;
  const top = mults.length ? Math.max(...mults) : 0;
  return { count: valid.length, realized, blended: blended.toFixed(1), top: top.toFixed(1) };
}
