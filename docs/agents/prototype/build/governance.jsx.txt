/* compact copiloted policy builder */
function PolicyBuilder({ pol, mobile, onBack, onAdopt }) {
  const [d, setD] = useState(() => ({ ...pol.rec }));
  const [applied, setApplied] = useState(false);
  const [phase, setPhase] = useState('edit');
  const set = (k, v) => setD(p => ({ ...p, [k]: v }));
  const toggle = (k, v) => setD(p => { const c = p[k] || []; return { ...p, [k]: c.includes(v) ? c.filter(x => x !== v) : [...c, v] }; });
  const steps = ['Pull your fund context', `Draft the ${pol.name.toLowerCase()}`, 'Check against the institutional standard', 'Prepare for adoption'];
  const [n, setN] = useState(0);
  useEffect(() => { if (phase !== 'building') return; let i = 0; const t = setInterval(() => { i += 1; setN(i); if (i >= steps.length) { clearInterval(t); setTimeout(() => setPhase('done'), 400); } }, 560); return () => clearInterval(t); }, [phase]);
  const rows = pol.decisions.map(dec => [dec.label, Array.isArray(d[dec.key]) ? (d[dec.key].join(', ') || 'None') : d[dec.key]]);

  if (phase === 'done') {
    return (
      <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', padding: '20px 0' }}>
        <div style={{ width: '100%', maxWidth: 540 }}>
          <div style={{ textAlign: 'center' }}>
            <div style={{ width: 54, height: 54, borderRadius: '50%', margin: '0 auto 13px', display: 'flex', alignItems: 'center', justifyContent: 'center', background: 'var(--success-soft)', border: '1px solid var(--success-line)' }}><Icon name="check" size={26} color="var(--success)" /></div>
            <div className="t-h2" style={{ fontSize: 19 }}>{pol.name} — adopted</div>
            <p style={{ fontSize: 12.5, color: 'var(--fg-3)', marginTop: 6 }}>Drafted to the institutional standard and active across the firm.</p>
          </div>
          <div style={{ borderRadius: 13, border: '1px solid var(--border)', overflow: 'hidden', marginTop: 16 }}>
            {rows.map(([k, v], i) => (
              <div key={k} style={{ display: 'flex', gap: 14, padding: '10px 15px', background: i % 2 ? 'transparent' : 'var(--surface-1)', borderTop: i ? '1px solid var(--border-faint)' : 'none' }}>
                <span style={{ width: 130, flex: 'none', fontSize: 12, color: 'var(--fg-4)' }}>{k}</span>
                <span style={{ fontSize: 13, fontWeight: 500, color: 'var(--fg-1)' }}>{v}</span>
              </div>
            ))}
          </div>
          <div style={{ display: 'flex', gap: 10, justifyContent: 'space-between', marginTop: 18, flexWrap: 'wrap' }}>
            <Button variant="ghost" icon="pencil" onClick={() => { setPhase('edit'); setN(0); }}>Go back &amp; edit</Button>
            <Button variant="gold" iconRight="check" onClick={onAdopt}>Adopt &amp; finish</Button>
          </div>
        </div>
      </div>
    );
  }
  if (phase === 'building') {
    return (
      <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', padding: '24px 0' }}>
        <div style={{ width: '100%', maxWidth: 520 }}>
          <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', textAlign: 'center', marginBottom: 16 }}>
            <div style={{ position: 'relative', marginBottom: 12 }}><div className="pulse-dot" style={{ position: 'absolute', inset: -10, borderRadius: '50%', background: 'radial-gradient(circle, rgba(247,201,72,0.5), transparent 70%)', filter: 'blur(8px)' }} /><div style={{ position: 'relative' }}><EarnCoin size={48} /></div></div>
            <div className="t-h2" style={{ fontSize: 18 }}>Drafting {pol.name}…</div>
          </div>
          <ProgressBar value={Math.round((n / steps.length) * 100)} gradient="linear-gradient(90deg,#F7C948,#E5A823)" height={6} />
          <div className="card" style={{ marginTop: 13, padding: 12, display: 'flex', flexDirection: 'column', gap: 2 }}>
            {steps.map((s, i) => i <= n && (
              <div key={i} className="fade-in" style={{ display: 'flex', alignItems: 'center', gap: 11, padding: '8px' }}>
                <span style={{ width: 22, height: 22, borderRadius: 999, flex: 'none', display: 'flex', alignItems: 'center', justifyContent: 'center', background: i < n ? 'var(--success-soft)' : 'var(--surface-2)', border: `1px solid ${i < n ? 'var(--success-line)' : 'var(--border)'}`, color: i < n ? 'var(--success)' : 'var(--fg-4)' }}>{i < n ? <Icon name="check" size={12} /> : <Icon name="loader" size={12} className="spin" />}</span>
                <span style={{ fontSize: 13, color: i < n ? 'var(--fg-2)' : 'var(--fg-1)' }}>{s}</span>
              </div>
            ))}
          </div>
        </div>
      </div>
    );
  }
  return (
    <div className="fade-in" style={{ display: 'flex', flexDirection: 'column', gap: 16 }}>
      <div style={{ display: 'flex', alignItems: 'center', gap: 12 }}>
        <button className="clickable" onClick={onBack} style={{ display: 'inline-flex', alignItems: 'center', gap: 6, fontSize: 12.5, color: 'var(--fg-3)', padding: '7px 11px', borderRadius: 9, border: '1px solid var(--border)', background: 'var(--surface-1)' }}><Icon name="arrow-left" size={14} />Policies</button>
        <div style={{ flex: 1, minWidth: 0 }}><div className="t-h2" style={{ fontSize: 18 }}>{pol.name}</div><div style={{ fontSize: 12, color: 'var(--fg-4)' }}>You set the standard, Earn drafts it</div></div>
        <Badge tone="azure" dot>Copiloted</Badge>
      </div>
      <div style={{ display: 'grid', gridTemplateColumns: mobile ? '1fr' : 'minmax(0,1fr) 300px', gap: 16, alignItems: 'start' }}>
        <div className="card" style={{ padding: 22 }}>
          <p style={{ fontSize: 12.5, color: 'var(--fg-4)', margin: '0 0 18px', lineHeight: 1.55 }}>{pol.intro}</p>
          <div style={{ display: 'flex', flexDirection: 'column', gap: 20 }}>
            {pol.decisions.map(dec => (
              <div key={dec.key}>
                <div className="t-eyebrow" style={{ marginBottom: 9 }}>{dec.label}{dec.kind === 'multi' && <span style={{ color: 'var(--fg-5)', fontWeight: 400, textTransform: 'none', letterSpacing: 0 }}> · pick any</span>}</div>
                <div style={{ display: 'flex', flexWrap: 'wrap', gap: 8 }}>
                  {dec.opts.map(o => <GovChip key={o} label={o} selected={dec.kind === 'multi' ? (d[dec.key] || []).includes(o) : d[dec.key] === o} onClick={() => dec.kind === 'multi' ? toggle(dec.key, o) : set(dec.key, o)} />)}
                </div>
              </div>
            ))}
          </div>
          <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center', marginTop: 24, gap: 10, flexWrap: 'wrap' }}>
            <Button variant="ghost" icon="arrow-left" onClick={onBack}>Cancel</Button>
            <Button variant="gold" iconRight="sparkles" onClick={() => { setN(0); setPhase('building'); }}>Draft &amp; adopt</Button>
          </div>
        </div>
        <div className="card" style={{ padding: 17, alignSelf: 'flex-start' }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: 10, marginBottom: 11 }}><div style={{ position: 'relative', flex: 'none' }}><EarnCoin size={32} /><span className="pulse-dot" style={{ position: 'absolute', right: -1, bottom: -1, width: 9, height: 9, borderRadius: 999, background: 'var(--success)', border: '2px solid var(--bg-2)' }} /></div><div><div style={{ fontSize: 13, fontWeight: 600 }}>Earn</div><div style={{ fontSize: 10.5, color: 'var(--fg-4)' }}>Compliance copilot</div></div></div>
          <div className="t-eyebrow" style={{ marginBottom: 7, color: 'var(--gold-1)' }}>Earn recommends</div>
          <div style={{ fontSize: 12.5, color: 'var(--fg-2)', lineHeight: 1.6 }}>{pol.recText}</div>
          <Button variant={applied ? 'secondary' : 'gold'} size="sm" icon={applied ? 'check' : 'sparkles'} style={{ width: '100%', marginTop: 14 }} onClick={() => { setD({ ...pol.rec }); setApplied(true); }}>{applied ? 'Recommendation applied' : "Apply Earn's recommendation"}</Button>
          <div style={{ display: 'flex', alignItems: 'center', gap: 6, marginTop: 12, fontSize: 11, color: 'var(--fg-5)' }}><Icon name="hand" size={12} />You're in control — change anything.</div>
        </div>
      </div>
    </div>
  );
}

  const [icIdx, setIcIdx] = useState(0);
  const [fm, setFm] = useState(FM_0);
  const [fmIdx, setFmIdx] = useState(0);
  const [adv, setAdv] = useState(ADV_0);
  const [advIdx, setAdvIdx] = useState(0);
  const [cap, setCap] = useState(CAP_0);
  const [capIdx, setCapIdx] = useState(0);
  const [legal, setLegal] = useState(LEGAL_0);
  const [legalIdx, setLegalIdx] = useState(0);
  const principal = mandate.principal || 'Jordan Avery';
  const firm = mandate.firm || 'Cedar Lane Capital';
  const adoptedCount = adopted.length;

  const isActive = (id) => adopted.includes(id);
  function adoptPolicy(id) {
    onAdopt && onAdopt(id, (GOV_POLICIES.find(p => p.id === id) || {}).name);
    setOpenPol(null);
  }
  function fillSeat() {
    const cand = IC_CANDIDATES[icIdx % IC_CANDIDATES.length];
    onRun({ id: 'ic-' + icIdx, hub: 'build', xp: 12, title: `Add ${cand.name} to the Investment Committee`,

Approve to extend the seat and set their voting rights.` } });

  function addManager() {

Approve to engage and add them to your counsel bench.` } });
  }
